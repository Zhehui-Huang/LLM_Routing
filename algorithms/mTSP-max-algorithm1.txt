\begin{itemize}    \item \( x_{ijk} \): Binary variable that is 1 if salesman \( k \) travels directly from node \( i \) to node \( j \), 0 otherwise.    \item \( u_i \): Continuous variable representing the position of node \( i \) in the tour.\end{itemize}\begin{itemize}    \item \( c_{ij} \): Cost (distance, time, etc.) associated with traveling from node \( i \) to node \( j \).    \item \( m \): Number of salesmen.    \item \( n \): Total number of nodes (including the depot).\end{itemize}Minimize the total travel cost:\[\min \max_{i=1, ...n, j=1, ..., n, k=1, ..., m} c_{ij} x_{ijk}\]\subsection*{Constraints}1. \textbf{Each city is visited exactly once by one salesman:}\[\sum_{i=1}^{n} \sum_{k=1}^{m} x_{ijk} = 1 \quad \forall j = 1, \ldots, n\]2. \textbf{Flow conservation constraints (each salesman must leave each visited node and enter each visited node exactly once):}\[\sum_{j=1}^{n} x_{ipk} - \sum_{j=1}^{n} x_{pjk} = 0 \quad \forall k = 1, \ldots, m, \; p = 1, \ldots, n\]3. \textbf{Each salesman must leave the depot exactly once:}\[\sum_{j=1}^{n} x_{1jk} = 1 \quad \forall k = 1, \ldots, m\]4. \textbf{Subtour elimination constraints:}\[u_i - u_j + n \sum_{k=1}^{m} x_{ijk} \leq n - 1 \quad \forall i \neq j, \; i, j = 2, \ldots, n\]5. \textbf{Binary constraints for assignment variables:}\[x_{ijk} \in \{0, 1\} \quad \forall i, j = 1, \ldots, n, \; k = 1, \ldots, m\]6. \textbf{Continuous variables for node positions:}\[u_i \geq 0 \quad \forall i = 2, \ldots, n\]
